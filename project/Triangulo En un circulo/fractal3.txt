import numpy as np
import matplotlib.pyplot as plt

def generar_fractal_limpio(c):
    # Configuración base del fractal
    width, height = 4096, 4096
    zoom_range = 2
    x = np.linspace(-zoom_range, zoom_range, width)
    y = np.linspace(-zoom_range, zoom_range, height)
    X, Y = np.meshgrid(x, y)
    Z = X + 1j * Y
    img = np.zeros(Z.shape)
    
    for i in range(128):
        mask = np.abs(Z) <= 2
        Z[mask] = Z[mask]**2 + c
        img[mask] += 1
    
    # Dibujo limpio
    fig, ax = plt.subplots(figsize=(8, 8))
    
    # Mostrar solo la imagen del fractal
    ax.imshow(img, cmap='magma', extent=[-zoom_range, zoom_range, -zoom_range, zoom_range], origin='lower')
    
    # Eliminar ejes, marcas y bordes
    ax.axis('off')
    
    # Eliminar espacios en blanco adicionales
    plt.subplots_adjust(left=0, right=1, top=1, bottom=0)
    
    plt.show()

# Parámetro del fractal
epsilon_geom = 1.240524959667
c = complex(epsilon_geom - 2, 0.1)

generar_fractal_limpio(c)
